lyr_list = []

for url in df2.URL:
    print('Now Scraping', url)
    
    response = requests.get(url)
    doc = BeautifulSoup(response.text)
    
    lyrics = doc.find_all('div', id='lyrics-body-text')
    for i in lyrics:
        lyr_dict = {}
        lyr_list2 = []
        for j in i.find_all('p'):
            lyr_list2.append(j.text)
        m = ' '.join(lyr_list2)
        lyr_dict['lyrics'] = m
    lyr_list.append(lyr_dict)
